import numpy as np
import matplotlib.pyplot as plt

plt.style.use("seaborn-v0_8")

x = np.linspace(0, 10, 100)
y = np.sin(x)

plt.figure()
plt.plot(x, y)
plt.title("Sine Wave Example")
plt.xlabel("X values")
plt.ylabel("sin(X)")
plt.tight_layout()
plt.show()
